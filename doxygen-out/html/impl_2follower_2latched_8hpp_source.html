46 template <typename DispatchT, typename LockPolicyT, typename ContainerT, typename QueueMonitorT>
47 State Latched<DispatchT, LockPolicyT, ContainerT, QueueMonitorT>::dry_capture_follower_impl(
48  const CaptureRange<stamp_type>& range)
49 {
50  if (PolicyType::queue_.empty())
51  {
52  if (latched_)
53  {
54  // If we have latched data, return that and report ready state
55  return State::PRIMED;
56  }
57  else
58  {
59  // Retry if queue has no data and there is no latched data
60  return State::RETRY;
61  }
62  }
63 
64  // The boundary before which data is are valid and after which they are not. Non-inclusive.
65  const stamp_type boundary = range.lower_stamp - min_period_;
66 
67  // Retry if priming is not possible
68  if (PolicyType::queue_.oldest_stamp() > boundary)
69  {
70  if (latched_)
71  {
72  // If we have latched data, return that and report ready state
73  return State::PRIMED;
74  }
75  else
76  {
77  // Abort if queue has no data and there is no latched data
78  return State::ABORT;
79  }
80  }
81 
82  // Find data closest to boundary, starting from oldest
83  // curr_qitr will never start at queue_.end() due to previous empty check
84  auto curr_qitr = PolicyType::queue_.begin();
85  auto prev_qitr = curr_qitr;
86  while (curr_qitr != PolicyType::queue_.end() and get_stamp(*curr_qitr) <= boundary)
87  {
88  prev_qitr = curr_qitr;
89  ++curr_qitr;
90  }
91 
92  // Set latched element
93  latched_ = *prev_qitr;
94 
95  // Remove all elements before latched element
96  const auto prev_stamp = get_stamp(*prev_qitr);
97  PolicyType::queue_.remove_before(prev_stamp);
98  return State::PRIMED;
99 }
